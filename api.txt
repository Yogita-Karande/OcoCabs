export const postData = async (data) => {
  try {
    data.api = 'login'
    const response = await axios.post(BASE_URL, data, {
      headers: {
        'Content-Type': 'application/json',
      },
    });
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
};
